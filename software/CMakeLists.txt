# Directory for PIO source files and the directory for generated headers
set(PIO_SRC_DIR ${CMAKE_CURRENT_SOURCE_DIR}/pio)
set(PIO_INC_DIR ${CMAKE_CURRENT_SOURCE_DIR}/include)

# Debug: Print PIO source and include directories
message(STATUS "PIO source directory: ${PIO_SRC_DIR}")
message(STATUS "PIO include directory: ${PIO_INC_DIR}")

# Automatically find all PIO files in the specified directory
file(GLOB PIO_SOURCES ${PIO_SRC_DIR}/*.pio)
message(STATUS "Found PIO files: ${PIO_SOURCES}")

# Command to compile PIO sources into header files
foreach(PIO_FILE ${PIO_SOURCES})
    get_filename_component(PIO_NAME ${PIO_FILE} NAME_WE)
    set(OUTPUT_FILE ${PIO_INC_DIR}/${PIO_NAME}.h)
    add_custom_command(
        OUTPUT ${OUTPUT_FILE}
        COMMAND pioasm ${PIO_FILE} ${OUTPUT_FILE}
        DEPENDS ${PIO_FILE}
        COMMENT "Compiling ${PIO_FILE}"
    )
    list(APPEND GENERATED_HEADERS ${OUTPUT_FILE})
    # Debug: Print each PIO file being processed
    message(STATUS "Scheduling compilation for: ${PIO_FILE} -> ${OUTPUT_FILE}")
endforeach()

# Inform CMake of the generated headers
add_custom_target(pio_headers DEPENDS ${GENERATED_HEADERS})

# Tell CMake where to find the executable source file
add_executable(${PROJECT_NAME} 
    main.cpp
    game.cpp
    display.cpp
    model_data.cpp
)

# Create map/bin/hex/uf2 files
pico_add_extra_outputs(${PROJECT_NAME})

# Link to pico_stdlib (gpio, time, etc. functions)
target_link_libraries(${PROJECT_NAME} 
    pico_stdlib
    hardware_dma
    pico-tflmicro
    hardware_pio
    hardware_irq
)

# Enable usb output, disable uart output
pico_enable_stdio_usb(${PROJECT_NAME} 1)
pico_enable_stdio_uart(${PROJECT_NAME} 0)

# Ensure the PIO assembly occurs before building the main executable
add_dependencies(${PROJECT_NAME} pio_headers)
